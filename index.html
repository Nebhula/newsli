function extractImage(it) { /* 1. thumbnail */ if (it.thumbnail && typeof it.thumbnail === 'string') { var t = it.thumbnail.trim(); if (t && t.indexOf('http') === 0) return t; } /* 2. enclosure */ if (it.enclosure && it.enclosure.url) { return it.enclosure.url; } /* 3. media:content */ if (it['media:content'] && it['media:content'].url) { return it['media:content'].url; } /* 4. content (extraer primera imagen del HTML) */ if (it.content) { var imgMatch = it.content.match(/]+src="([^">]+)"/); if (imgMatch && imgMatch[1]) { return imgMatch[1]; } } /* 5. description (extraer primera imagen del HTML) */ if (it.description) { var imgMatch = it.description.match(/]+src="([^">]+)"/); if (imgMatch && imgMatch[1]) { return imgMatch[1]; } } /* 🔥 SIN FAVICON. SIN PLACEHOLDER. Si no hay imagen, devuelve null */ return null; }
